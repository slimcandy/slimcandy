1. Fill in package.json with the following:

   ```json
   {
     "description": "Project description",
     "author": "NICKNAME",
     "keywords": ["keywords", "matching", "your", "project"],
     "homepage": "https://WEBSITE.COM"
   }
   ```

2. Install [Helmet](https://www.npmjs.com/package/react-helmet)

   `npm install --save react-helmet`

   ```tsx
   import { Helmet } from 'react-helmet'
   import packageJson from './path/to/package.json'

   function HelmetComponent() {
     const title = 'Random AR'
     const { description, homepage: url } = packageJson

     return (
       <Helmet>
         <title>{title}</title>
         <meta name="description" content={description} />
         <link rel="canonical" href={url} />

         <meta property="og:url" content={url} />
         <meta property="og:type" content="website" />
         <meta property="og:title" content={title} />
         <meta property="og:description" content={description} />
       </Helmet>
     )
   }
   ```

3. Put `<HelmetComponent />` into root component.
